Understanding Missing Data Strategies

Strategy Best For Caution
Deletion Small % of missing data, random pattern Can introduce bias, lose information
Mean/Median Numeric data, normal distribution Reduces variance, not for categorical
Mode Categorical data, clear majority class Can skew distribution
Forward/Backward Fill Time series, sequential data Assumes continuity
Interpolation Numeric time series with trends May not capture seasonality

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import mysql.connector
import psycopg2
from datetime import datetime

sns.set_style("whitegrid")

def load_dirty_data():
    """Load data from databases"""
    mysql_conn = mysql.connector.connect(
        host="localhost", user="root", 
        password="rootpassword", database="sales_data_raw"
    )
    pg_conn = psycopg2.connect(
        host="localhost", user="postgres", 
        password="postgrespassword", database="employee_data_raw"
    )
    
    customers = pd.read_sql("SELECT * FROM customers", mysql_conn)
    products = pd.read_sql("SELECT * FROM products", mysql_conn)
    employees = pd.read_sql("SELECT * FROM employees", pg_conn)
    
    mysql_conn.close()
    pg_conn.close()
    
    return customers, products, employees

def handle_missing_customers(df):
    """Apply appropriate strategies for customer data"""
    df_clean = df.copy()
    
    # Strategy 1: Fill missing emails with placeholder
    df_clean['email'] = df_clean['email'].fillna('no-email@unknown.com')
    
    # Strategy 2: Fill missing phone with placeholder
    df_clean['phone'] = df_clean['phone'].fillna('Unknown')
    
    # Strategy 3: Fill missing age with median (robust to outliers)
    median_age = df_clean['age'][(df_clean['age'] > 0) & (df_clean['age'] < 120)].median()
    df_clean['age'] = df_clean['age'].fillna(median_age)
    
    # Strategy 4: Fill missing balance with 0 (new accounts)
    df_clean['account_balance'] = df_clean['account_balance'].fillna(0.0)
    
    # Strategy 5: Fill missing registration date with most recent date
    df_clean['registration_date'] = df_clean['registration_date'].fillna('2024-01-01')
    
    # Strategy 6: Fill missing names with 'Unknown Customer'
    df_clean['name'] = df_clean['name'].fillna('Unknown Customer')
    
    return df_clean

def handle_missing_products(df):
    """Apply appropriate strategies for product data"""
    df_clean = df.copy()
    
    # Fill missing category with 'Uncategorized'
    df_clean['category'] = df_clean['category'].fillna('Uncategorized')
    
    # Fill missing stock with 0
    df_clean['stock'] = df_clean['stock'].fillna(0)
    
    # Fill missing supplier with 'Unknown Supplier'
    df_clean['supplier'] = df_clean['supplier'].replace('', 'Unknown Supplier')
    df_clean['supplier'] = df_clean['supplier'].fillna('Unknown Supplier')
    
    # Fill missing last_updated with current date
    df_clean['last_updated'] = df_clean['last_updated'].fillna('2024-01-01')
    
    # Fill missing product_name with 'Unknown Product'
    df_clean['product_name'] = df_clean['product_name'].fillna('Unknown Product')
    
    return df_clean

def handle_missing_employees(df):
    """Apply appropriate strategies for employee data"""
    df_clean = df.copy()
    
    # Fill missing names with 'Unknown'
    df_clean['first_name'] = df_clean['first_name'].fillna('Unknown')
    df_clean['last_name'] = df_clean['last_name'].fillna('Unknown')
    
    # Fill missing email with generated placeholder
    def generate_email(row):
        if pd.isna(row['email']):
            first = str(row['first_name']).lower()
            last = str(row['last_name']).lower()
            return f"{first}.{last}@company.com"
        return row['email']
    
    df_clean['email'] = df_clean.apply(generate_email, axis=1)
    
    # Fill missing department with 'Unassigned'
    df_clean['department'] = df_clean['department'].replace('', 'Unassigned')
    df_clean['department'] = df_clean['department'].fillna('Unassigned')
    
    # Fill missing salary with median by department
    for dept in df_clean['department'].unique():
        dept_median = df_clean[df_clean['department'] == dept]['salary'].median()
        df_clean.loc[(df_clean['department'] == dept) & (df_clean['salary'].isna()), 'salary'] = dept_median
    
    # Fill remaining missing salary with overall median
    df_clean['salary'] = df_clean['salary'].fillna(df_clean['salary'].median())
    
    # Fill missing hire_date with median date
    df_clean['hire_date'] = df_clean['hire_date'].fillna('2023-01-01')
    
    # Fill missing performance_score with department average
    for dept in df_clean['department'].unique():
        dept_avg = df_clean[df_clean['department'] == dept]['performance_score'].mean()
        df_clean.loc[(df_clean['department'] == dept) & (df_clean['performance_score'].isna()), 'performance_score'] = dept_avg
    
    df_clean['performance_score'] = df_clean['performance_score'].fillna(df_clean['performance_score'].mean())
    
    return df_clean

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import re
from datetime import datetime

sns.set_style("whitegrid")

def standardize_dates(date_str):
    """Convert various date formats to standard YYYY-MM-DD"""
    if pd.isna(date_str):
        return date_str
    
    date_str = str(date_str).strip()
    
    # Try different date formats
    formats = [
        '%Y-%m-%d',      # 2024-01-15
        '%d/%m/%Y',      # 15/01/2024
        '%m/%d/%Y',      # 01/15/2024
        '%Y/%m/%d',      # 2024/01/15
    ]
    
    for fmt in formats:
        try:
            date_obj = datetime.strptime(date_str, fmt)
            return date_obj.strftime('%Y-%m-%d')
        except ValueError:
            continue
    
    return date_str

def standardize_text(text):
    """Normalize text: lowercase, strip whitespace"""
    if pd.isna(text):
        return text
    text = str(text).strip().lower()
    text = re.sub(r'\s+', ' ', text)  # Replace multiple spaces
    return text

def standardize_phone(phone):
    """Standardize phone numbers to +XX-XXX-XXXX format"""
    if pd.isna(phone):
        return phone
    
    # Remove all non-numeric characters except +
    phone = str(phone)
    digits = re.sub(r'[^\d+]', '', phone)
    
    # If it starts with +, keep it; otherwise add +1
    if not digits.startswith('+'):
        digits = '+1' + digits
    
    return digits

def standardize_customers(df):
    """Apply standardization to customer data"""
    df_clean = df.copy()
    
    # Standardize dates
    df_clean['registration_date'] = df_clean['registration_date'].apply(standardize_dates)
    
    # Standardize text fields
    df_clean['name'] = df_clean['name'].apply(standardize_text)
    df_clean['email'] = df_clean['email'].str.lower().str.strip()
    df_clean['country'] = df_clean['country'].apply(standardize_text)
    
    # Standardize phone numbers
    df_clean['phone'] = df_clean['phone'].apply(standardize_phone)
    
    # Fix invalid ages (negative or > 120)
    df_clean.loc[(df_clean['age'] < 0) | (df_clean['age'] > 120), 'age'] = df_clean['age'].median()
    
    # Fix negative balances (set to 0)
    df_clean.loc[df_clean['account_balance'] < 0, 'account_balance'] = 0.0
    
    return df_clean

def standardize_products(df):
    """Apply standardization to product data"""
    df_clean = df.copy()
    
    # Standardize dates
    df_clean['last_updated'] = df_clean['last_updated'].apply(standardize_dates)
    
    # Standardize text
    df_clean['product_name'] = df_clean['product_name'].apply(standardize_text)
    df_clean['category'] = df_clean['category'].apply(standardize_text)
    df_clean['supplier'] = df_clean['supplier'].str.strip()
    
    # Fix negative prices
    df_clean.loc[df_clean['price'] < 0, 'price'] = df_clean['price'].median()
    
    # Fix negative stock
    df_clean.loc[df_clean['stock'] < 0, 'stock'] = 0
    
    return df_clean

def standardize_employees(df):
    """Apply standardization to employee data"""
    df_clean = df.copy()
    
    # Standardize dates
    df_clean['hire_date'] = df_clean['hire_date'].apply(standardize_dates)
    
    # Standardize text
    df_clean['first_name'] = df_clean['first_name'].apply(standardize_text)
    df_clean['last_name'] = df_clean['last_name'].apply(standardize_text)
    df_clean['email'] = df_clean['email'].str.lower().str.strip()
    df_clean['department'] = df_clean['department'].apply(standardize_text)
    
    # Fix negative salaries
    df_clean.loc[df_clean['salary'] < 0, 'salary'] = df_clean['salary'].median()
    
    # Fix invalid performance scores (must be 0-5)
    df_clean.loc[(df_clean['performance_score'] < 0) | (df_clean['performance_score'] > 5), 'performance_score'] = df_clean['performance_score'].median()
    
    return df_clean

def remove_duplicates(df, subset=None):
    """Remove duplicate records based on key columns"""
    before_count = len(df)
    df_clean = df.drop_duplicates(subset=subset, keep='first')
    after_count = len(df_clean)
    removed = before_count - after_count
    
    return df_clean, removed

Outlier Detection Methods

Method Description Best For
IQR (Interquartile Range) Values < Q1 - 1.5*IQR or > Q3 + 1.5*IQR General purpose, non-normal distributions
Z-Score |z-score| > 3 (or 2.5) Normal distributions
Domain Knowledge Business rules (e.g., age > 120) When business context is clear 
# outlier_detection.py

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from scipy import stats

sns.set_style("whitegrid")

def detect_outliers_iqr(df, column):
    """Detect outliers using IQR method"""
    Q1 = df[column].quantile(0.25)
    Q3 = df[column].quantile(0.75)
    IQR = Q3 - Q1
    
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    
    outliers = df[(df[column] < lower_bound) | (df[column] > upper_bound)]
    
    return outliers, lower_bound, upper_bound

def treat_outliers(df, column, method='cap'):
    """
    Treat outliers using specified method
    Methods: 'cap' (winsorize), 'remove', 'median'
    """
    df_clean = df.copy()
    
    # Detect outliers using IQR
    Q1 = df_clean[column].quantile(0.25)
    Q3 = df_clean[column].quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    
    if method == 'cap':
        # Cap outliers at bounds (winsorization)
        df_clean[column] = df_clean[column].clip(lower=lower_bound, upper=upper_bound)
    elif method == 'remove':
        # Remove outlier rows
        df_clean = df_clean[(df_clean[column] >= lower_bound) & (df_clean[column] <= upper_bound)]
    elif method == 'median':
        # Replace outliers with median
        median = df_clean[column].median()
        df_clean.loc[(df_clean[column] < lower_bound) | (df_clean[column] > upper_bound), column] = median
    
    return df_clean, lower_bound, upper_bound
